2. Better example


def pysum(start, step, count):
    ret = start
    for i in range(count):
        ret += step
    return ret

_mod.pyx


try:
    from ._mod import _append
except ImportError:
    _append = None

def _pyappend(lst, item, count):
    for i in range(count):
        lst.append(item)


if _append is not None:
    append = _append
else:
    append = _pyappend
